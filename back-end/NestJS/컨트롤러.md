Express의 Router와 같음.<br>
URL 받아와서 실행할 함수를 매핑함.

```js
@Controller()
export class AppController {
  constructor(private readonly appService: AppService) {}

  @Get('/hello')
  getHello(): string {
    return this.appService.getHello();
  }
  @Get('/hi')
  getHi(): string {
    return this.appService.getHi();
  }
}
```
